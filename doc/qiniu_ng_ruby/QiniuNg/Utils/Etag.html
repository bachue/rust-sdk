def from_io(io)
  io.binmode
  e = Bindings::Etag.new!
  e.update(io.read(1 << 22)) until io.eof?
  etag_result = FFI::MemoryPointer::new(ETAG_SIZE)
  e.result(etag_result)
  etag_result.read_bytes(ETAG_SIZE)
end
